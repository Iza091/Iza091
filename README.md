  ```javascript
class Coffee {
  constructor() {
    this.empty = true;
  }

  drink() {
    if (!this.empty) {
      console.log("You have taken a sip of coffee. Enjoy!");
    } else {
      console.log("Your coffee cup is empty. Please refill it.");
    }
  }

  refill() {
    this.empty = false;
    console.log("Your coffee cup has been refilled. Enjoy!");
  }
}

const coffee = new Coffee();

if (coffee.empty) {
  coffee.refill();
} else {
  coffee.drink();
}
``` 
